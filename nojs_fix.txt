! ====================================== Generic srcset fixes ======================================
! techradar-sites
gamesradar.com,techradar.com,tomsguide.com,tomshardware.com,whathifi.com,windowscentral.com##+js(srcset, img.lazy-image-van, srcset, 1)

! techxplore-sites
medicalxpress.com,phys.org,techxplore.com##+js(srcset, img[src^="data:image"], src)

! mangabtt.com
mangabtt.com##+js(srcset, .detail-info img.lazyload, src)

! mysmartprice.com
mysmartprice.com##+js(srcset, img[data-lazy-src], lazySrc)
mysmartprice.com##img[data-lazy-fallback]:style(opacity: 1 !important;)

! scitechdaily.com
scitechdaily.com##+js(srcset, img.ezlazyload, ezsrc, 1)

! usatoday.com
usatoday.com##+js(srcset, img[data-g-r^="lazy"], glSrcset, 1)


! ====================================== Specific sites-fix ======================================
! businessinsider.com
businessinsider.com##+js(bi-fix)
businessinsider.com##img.lazy-image:style(opacity: 1 !important;)

! insideevs.com
insideevs.com##+js(ie-fix)
insideevs.com##.detailed-box .lazyload

! vice.com
vice.com##+js(vice-fix)

! windowscentral.com
windowscentral.com##+js(windowscentral-fix)
